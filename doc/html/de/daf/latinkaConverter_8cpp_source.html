196 std::wstring Latinka::Converter::translitterateLetter(const Alphabet& from, const std::wstring &c1, const std::wstring &c2) const
197 {
198  std::wstring res;
199  if (from == Alphabet::Cyrillic){
200  if (c2 == L"в")
201  return res += isLower(c1) ? L"w" : L"W";
202  else {
203  res += translitterateLetter(from, c1);
204  if (c2 != L"\'" && getLetterType(c1) != LType::NoSoftly)
205  res += L"\u030C";
206 
207  if (getLetterType(c2) == LType::DoubleLoud){
208  auto letter = translitterateLetter(from, c2);
209  res += letter[1];
210  }
211  }
212  }
213  else if (from == Alphabet::Latin){
214  if (c1.size() == 2){ // for soft letter and loud, it convert to consonantal + doubleLoud
215  res += translitterateLetter(from, c1[0]);
216  res += translitterateLetter(from, L'j' + c2);
217  }
218  else if (c1 == L"j" || c1 == L"J") // doubleLoud Letter
219  res += translitterateLetter(from, c1 + c2);
220  else if (c2 == L"\u030C" && getLetterType(c1 + c2) == Letter::NoSoftly)
221  res += translitterateLetter(from, c1 + c2);
222  else if (c2 == L"\u030C"){
223  res += translitterateLetter(from, c1);
224  res += L'ь';
225  }
226  else if (c2 == L"ć") // ść to щ
227  res += translitterateLetter(from, c1 + c2);
228  }
229 
230  return res;
231 }
